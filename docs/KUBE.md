
```shell
brew install kubernetes-cli

brew install aws

```

# exec into a pod

```shell
kubectl exec <pod-name> <command> --namespace=<namespace>
kubectl exec prometheus-op-kube-state-metrics-85b7ffcb89-s96z7 /bin/bash --namespace=monitoring
```
